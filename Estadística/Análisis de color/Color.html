# Make radio button less cramped by adding a space after each label
# The spacing will only show up in your IDE, not on this doc page
options = ['Central', 'East', 'South', 'West']
labels = [option + ' ' for option in options]

input_dropdown = alt.binding_radio(
    # Add the empty selection which shows all when clicked
    options=options + [None],
    labels=labels + ['All'],
    name='Region: '
)
selection = alt.selection_point(
    fields=['Region'],
    bind=input_dropdown,
)

alt.Chart(df).mark_point().encode(
    x='Sales:Q',
    y='Profit:Q',
    # We need to set a constant domain to preserve the colors
    # when only one region is shown at a time
    color=alt.Color('Region:N').scale(range=logo),
).add_params(
    selection
).transform_filter(
    selection
).interactive()

    value='Category',
    bind=dropdown
)

    x=alt.X('x:N').title(''),

    color=alt.Color('Segment:N', title='Segment', scale=alt.Scale(range=cuadrados)),
    tooltip=['Category:N', 'Region:N', 'Sub-Category:N', 'Sales:Q']
).transform_calculate(
    x=f'datum[{xcol_param.name}]'
).add_params(
    xcol_param
).properties(
    title='Sales by Selected X-axis'
).interactive()

# Mostrar la grfica
chart.show()

# Paletas de colores (ejemplo)
triada = ["#F23DC7", "#3DF2F0", "#F2D43D", "#9D925C", "#73566C"]
cuadrados = ["#F2493D", "#443DF2", "#F2D43D", "#3DF289", "#9D925C"]
logo = ["#F22233", "#F2E527", "#F2D43D", "#F2A03D", "#D93814"]
monocromatico = ["#BFAE56", "#8C845B", "#F2D43D", "#59574C", "#333126"]

# Grfica de ventas con men desplegable
dropdown = alt.binding_select(
    options=['Category', 'Region', 'Sub-Category'],
    name='X-axis column '
)
xcol_param = alt.param(
    value='Category',
    bind=dropdown
)

sales_chart = alt.Chart(df).mark_circle(size=60).encode(
    x=alt.X('x:N').title(''),
    y=alt.Y('Sales:Q', title='Sales'),
    color=alt.Color('Segment:N', title='Segment', scale=alt.Scale(range=cuadrados)),
    tooltip=['Category:N', 'Region:N', 'Sub-Category:N', 'Sales:Q']
).transform_calculate(
    x=f'datum[{xcol_param.name}]'
).add_params(
    xcol_param
).properties(
    title='Sales by Selected X-axis',
    width=500,
    height=300
).interactive()

# Grfica de punto con filtro de regin
options = ['Central', 'East', 'South', 'West']
labels = [option + ' ' for option in options]

input_dropdown = alt.binding_radio(
    options=options + [None],
    labels=labels + ['All'],
    name='Region: '
)
selection = alt.selection_point(
    fields=['Region'],
    bind=input_dropdown,
)

scatter_plot = alt.Chart(df).mark_point().encode(
    x='Sales:Q',
    y='Profit:Q',
    color=alt.Color('Region:N', scale=alt.Scale(range=logo)),
    tooltip=['Region:N', 'Sales:Q', 'Profit:Q']
).add_params(
    selection
).transform_filter(
    selection
).properties(
    title='Sales and Profit by Region',
    width=500,
    height=300
).interactive()

# Pie chart of region
pie_chart = alt.Chart(df).mark_arc(innerRadius=50).encode(
    theta=alt.Theta(field='count', type='quantitative'),
    color=alt.Color("Region:N", scale=alt.Scale(range=monocromatico)),
    tooltip=['Region:N', alt.Tooltip('count()', title='Count')]
).properties(
    title='Region',
    width=300,
    height=300
).transform_aggregate(
    count='count()',
    groupby=['Region']
)

# Bar chart of category and count
bar_chart_category = alt.Chart(df).mark_bar().encode(
    x=alt.X('Category:N', title='Category'),
    y=alt.Y('count()', title='Count'),
    color=alt.Color("Category:N", scale=alt.Scale(range=logo), legend=None),
    tooltip=['Category:N', alt.Tooltip('count()', title='Count')]
).properties(
    title='Category Count',
    height=400,
    width=600
)

# Concatenar todas las grficas en un solo tablero
final_dashboard = alt.vconcat(
    sales_chart,
    alt.hconcat(
        scatter_plot,
        pie_chart,
        bar_chart_category
    )
).resolve_scale(
    color='independent'
).configure_mark(
    cornerRadius=3
).configure_view(
    strokeWidth=0
).configure_title(
    fontSize=20,
    anchor='start',
    color='gray'
)

# Mostrar la grfica
final_dashboard.show()

El tablero interactivo combina mltiples visualizaciones para analizar las ventas de una empresa desde diferentes perspectivas. Incluye una grfica de barras que muestra las ventas por categora, una grfica de burbujas con un men desplegable que permite seleccionar diferentes dimensiones del eje X (Categora, Regin y Sub-Categora), una grfica de dispersin que filtra por regin, un grfico de pastel que representa la distribucin de ventas por regin, y una grfica de barras que muestra el conteo de productos por categora. Las paletas de colores variadas facilitan la diferenciacin visual de las categoras y regiones.

Conclucin

Este tablero proporciona una vista integral de las ventas y el rendimiento de diferentes categoras y regiones, facilitando la identificacin de tendencias y reas clave de mejora. La interactividad y los filtros permiten una exploracin detallada de los datos, haciendo ms accesible la toma de decisiones informadas.
