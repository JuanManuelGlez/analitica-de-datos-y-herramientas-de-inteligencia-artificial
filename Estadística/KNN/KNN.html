#Sacamos la distancia euclidiana entre dos puntos y lo ordenamos de menor a mayor

def euclidean_distance(row1, row2):
    distance = 0.0
    for i in range(len(row1)-1):
        distance += (row1[i] - row2[i])**2
    return math.sqrt(distance)

#Se ordenan las instancias basadas en su distancia a cada patron de prueba 

def get_neighbors(train , test_row, num_neighbors): # train es el conjunto de entrenamiento, test_row es el patron de prueba
    distances = list()
    for train_row in train:
        dist = euclidean_distance(test_row, train_row)
        distances.append((train_row, dist))
    distances.sort(key=lambda tup: tup[1]) 
    neighbors = list()
    for i in range(num_neighbors):
        neighbors.append(distances[i][0])

    #imprimimos neighbors

    #print(neighbors)
    return neighbors

#Se obtiene la clase de cada vecino y se elige la clase que mas se repite

def predict_classification(train, test_row, num_neighbors):
    neighbors = get_neighbors(train, test_row, num_neighbors)
    output_values = [row[0] for row in neighbors]
    prediction = max(set(output_values), key=output_values.count)
    return prediction
